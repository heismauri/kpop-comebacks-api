const currentUserTimeZone = Intl.DateTimeFormat().resolvedOptions().timeZone;
const headerBoxWrapper = document.getElementById('header-box-wrapper');
const countryElement = document.getElementById('country');

const isFromTheUS = () => timezoneCountries[currentUserTimeZone] === 'US';

const countryToEmoji = (c) => c.replace(/./g, (ch) => String.fromCodePoint(0x1f1a5 + ch.toUpperCase().charCodeAt()));

const printFlagAndTimeZone = async () => {
  countryElement.innerHTML = `<p>${currentUserTimeZone}<strong class="flag">${countryToEmoji(timezoneCountries[currentUserTimeZone])}</strong></p>`;
  headerBoxWrapper.classList.add('opacity-100');
};

printFlagAndTimeZone();
